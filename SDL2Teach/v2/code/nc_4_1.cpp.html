#include <SDL2/SDL.h>
#include <SDL2/SDL_ttf.h>
// 窗口
SDL_Window *Window = NULL;
//渲染器
SDL_Renderer *Renderer = NULL;
//纹理贴图
SDL_Texture *Texture = NULL;
//表面
//SDL_Surface *Surface = NULL;


int Init()
{
  SDL_Init(SDL_INIT_VIDEO);
  // 创建窗口
  Window = SDL_CreateWindow("SDL_Window", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, 240, 400, SDL_WINDOW_SHOWN);
  if (Window == NULL)
    return -1;
  //创建渲染器
  Renderer = SDL_CreateRenderer(Window, -1, SDL_RENDERER_ACCELERATED);
  if (Renderer == NULL)
    return -2;
  return 0;
}



void Destroy()
{

  SDL_DestroyWindow(Window);
  SDL_DestroyTexture(Texture);
  SDL_DestroyRenderer(Renderer);
  SDL_Quit();
}

 /*******本课重点********/

SDL_Texture *Text(const char *Fonts, int FontSize, SDL_Renderer * Render, const char *FontsPath, int r = 0, int g = 0, int b = 0, int a = 0)
{

  // 初始化字体
  TTF_Init();

  TTF_Font *font = NULL;

//打开字体/* 字体路径 /*字体大小
  font = TTF_OpenFont(FontsPath, FontSize);

  /*
   *结构体原型
   SDL_Color
   {
   Uint8 r,g,b,a;
   };
   */
  //字体颜色
  SDL_Color color;

  color.r = r;
  color.g = g;
  color.b = b;
  color.a = a;

  SDL_Surface *temp = NULL;
  SDL_Texture *fonts = NULL;
  // 绘制表面……字体编码为 UTF-8
  temp = TTF_RenderUTF8_Blended(font, Fonts, color);
  //创建贴图
  fonts = SDL_CreateTextureFromSurface(Render, temp);
  SDL_FreeSurface(temp);
  //释放字体
  TTF_CloseFont(font);
  TTF_Quit();
  if (fonts != NULL)
    return fonts;

}

int main(int args, char *argv[])
{
  //初始化
  if (Init() != 0)
    return -1;

 /*******本课重点********/
  //粘贴用的矩形参数
  SDL_Rect Box;
#if defined(__ANDROID__)
//安卓平台使用手机本身的字体
  Texture = Text("DXkite字体测试", 28, Renderer, "/system/fonts/DroidSansFallback.ttf", 0xff);
#else
//其他平台使用默认字体
  Texture = Text("DXkite", 28, Renderer, "f.ttf", 0xff);
#endif              /*  */
  //粘贴表面的盒子
  Box.x = 0;
  Box.y = 0;
  /*
   *查询函数
   *int SDL_QueryTexture(SDL_Texture* texture,Uint32* format, int* access,int* w,int* h)
   用于查询 Texture的一些属性 
   format 像素格式
   Access 
   w weight 宽度 
   h height 高度
   
   */
  int w = 0, h = 0;
  SDL_QueryTexture(Texture, NULL, NULL, &w, &h);
  Box.w = w;
  Box.h = h;

  //清理渲染器
  SDL_RenderClear(Renderer);
  //复制绘图后的表面至渲染器 
  SDL_RenderCopy(Renderer, Texture, NULL, &Box);
//刷新
  SDL_RenderPresent(Renderer);
  SDL_Delay(2000);
  Destroy();
  return 0;
}
