#include <SDL2/SDL.h>
#include <SDL2/SDL_mixer.h>
// 窗口
SDL_Window *Window = NULL;
//渲染器
SDL_Renderer *Renderer = NULL;
//纹理贴图
SDL_Texture *Texture = NULL;

// 图片表面
SDL_Surface *Surface = NULL;

//Mix_Chunk *Chunk[5] = NULL;



class player {
  Mix_Chunk *Chunk[4];
public:
   player() {
    init();

  }
  ~player() {
    exit();

  }
  void init() {
    //Mix_Init(int flags);
    //extern DECLSPEC int SDLCALL Mix_OpenAudio(int frequency, Uint16 format, int channels, int chunksize);
    Mix_OpenAudio(MIX_DEFAULT_FREQUENCY, MIX_DEFAULT_FORMAT, MIX_DEFAULT_CHANNELS, 2048);
    for (int i = 0; i < 4; ++i)
      Chunk[i] = NULL;

  } int load() {
    Chunk[0] = Mix_LoadWAV("Gunfire.wav");
    Chunk[1] = Mix_LoadWAV("Bang.wav");
    Chunk[2] = Mix_LoadWAV("Fanfare.wav");
    Chunk[3] = Mix_LoadWAV("peow.wav");
    for (int i = 0; i < 4; ++i) {
      if (Chunk[i] == NULL)
    return i;

    }
  }
  void play(int i) {
    //extern DECLSPEC int SDLCALL Mix_PlayChannelTimed(int channel, Mix_Chunk *chunk, int loops, int ticks);
    /* Play an audio chunk on a specific channel.
       If the specified channel is -1, play on the first free channel.
       If 'loops' is greater than zero, loop the sound that many times.
       If 'loops' is -1, loop inifinitely (~65000 times).
       Returns which channel was used to play the sound.
     */

    if (i > 3 || i < 0)
      return;
    Mix_PlayChannelTimed(1, Chunk[i], 0, -1);
  }
  void exit() {
    for (int i = 0; i < 4; ++i)
      Mix_FreeChunk(Chunk[i]);
    Mix_Quit();
  }
};

int Init()
{
  SDL_Init(SDL_INIT_VIDEO);
  // 创建窗口
  Window = SDL_CreateWindow("SDL_Window", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, 240, 400, SDL_WINDOW_SHOWN);
  if (Window == NULL)
    return -1;
  //创建渲染器
  Renderer = SDL_CreateRenderer(Window, -1, SDL_RENDERER_ACCELERATED);
  if (Renderer == NULL)
    return -2;
  return 0;
}

void Destroy()
{
  SDL_FreeSurface(Surface);
  SDL_DestroyWindow(Window);
  SDL_DestroyTexture(Texture);
  SDL_DestroyRenderer(Renderer);
  SDL_Quit();
}


int main(int args, char *argv[])
{
  //初始化
  if (Init() != 0)
    return -1;
  //状态标志
  bool Quit = false;
//事件储存器
  SDL_Event Event;
  player play;
  play.load();

  //获取键盘焦点
  Window = SDL_GetKeyboardFocus();

  //打开键盘(手机专用)
  SDL_StartTextInput();
  while (!Quit) {

    //将事件储存至储存器中
    while (SDL_PollEvent(&Event) != 0) {

      if (Event.type == SDL_KEYDOWN)
    switch (Event.key.keysym.sym) {
    case SDLK_UP:
      {
        //😊Play();
        play.play(0);
        break;
      }
    case SDLK_DOWN:
      {
        //Play();
        play.play(1);
        break;
      }
    case SDLK_LEFT:
      {
        //Play();
        play.play(2);
        break;
      }
    case SDLK_RIGHT:
      {
        //   Play();
        play.play(3);
        break;
      }
      //菜单键退出……
    case SDLK_MENU:{
        // dbg("MENU");
        Quit = true;
        break;
      }
    }
    }
  }
  Destroy();
  return 0;
}
